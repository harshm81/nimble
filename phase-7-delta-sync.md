model SyncLogDailySummary {
  id                  BigInt   @id @default(autoincrement()) @db.UnsignedBigInt
  date                DateTime @db.DateTime(3)

  jobType             String   @map("job_type") @db.VarChar(100)

  successCount        Int      @map("success_count")
  failureCount        Int      @map("failure_count")
  totalRecordsFetched Int?     @map("total_records_fetched")
  totalRecordsSaved   Int?     @map("total_records_saved")
  totalRecordsFailed  Int?     @map("total_records_failed")
  avgDurationMs       Int?     @map("avg_duration_ms")
  maxDurationMs       Int?     @map("max_duration_ms")
  createdAt           DateTime @default(now()) @map("created_at") @db.DateTime(3)

  @@unique([date, platform, jobType], name: "date_platform_job_type")

### Step 4 — Repository `src/db/repositories/syncSummaryRepo.ts`

Write interfaces by reading the Prisma model field names. Use `created_at` (not `started_at` — that column does not exist in `sync_logs`).

```ts
import prisma from '../prismaClient';

export async function writeDailySummary(date: Date): Promise<void> {
  await prisma.$executeRaw`
    INSERT INTO sync_log_daily_summary
      (date, platform, job_type, total_runs, success_count, failure_count,
       total_records_fetched, total_records_saved, total_records_failed,
       avg_duration_ms, max_duration_ms)
    SELECT
      DATE(created_at)            AS date,
      platform,
      job_type,
      COUNT(*)                    AS total_runs,
      SUM(status = 'success')     AS success_count,
      SUM(status = 'failed')      AS failure_count,
      SUM(records_fetched)        AS total_records_fetched,
      SUM(records_saved)          AS total_records_saved,
      SUM(records_failed)         AS total_records_failed,
      ROUND(AVG(duration_ms))     AS avg_duration_ms,
      MAX(duration_ms)            AS max_duration_ms
    FROM sync_logs
    WHERE DATE(created_at) = DATE(${date}) - INTERVAL 1 DAY
    GROUP BY DATE(created_at), platform, job_type
    ON DUPLICATE KEY UPDATE
      total_runs           = VALUES(total_runs),
      success_count        = VALUES(success_count),
      failure_count        = VALUES(failure_count),
      total_records_fetched = VALUES(total_records_fetched),
      total_records_saved  = VALUES(total_records_saved),
      total_records_failed = VALUES(total_records_failed),
      avg_duration_ms      = VALUES(avg_duration_ms),
      max_duration_ms      = VALUES(max_duration_ms)
  `;
}

export async function cleanupOldLogs(): Promise<number> {
  return prisma.$executeRaw`
    DELETE FROM sync_logs WHERE created_at < NOW() - INTERVAL 90 DAY
  `;
}
```

> Column names in `$executeRaw` use `snake_case` to match the DB schema. `created_at` maps to the `SyncLog.createdAt` Prisma field — this is the timestamp when the job was queued (i.e. the row insert time). There is no `started_at` column.

### Step 5 — Worker `src/workers/maintenanceWorker.ts`

Follow the same worker pattern as all other workers. Maintenance jobs are housekeeping — no `getLastSyncedAt` / `setLastSyncedAt` needed, no sync cursor to advance.

```ts
import { Worker } from 'bullmq';
import { connection } from '../queue/connection';
import { MAINTENANCE_PLATFORM, MAINTENANCE_QUEUE, MAINTENANCE_JOBS } from '../constants/maintenance';
import { logger } from '../utils/logger';
import { writeDailySummary, cleanupOldLogs } from '../db/repositories/syncSummaryRepo';

export const maintenanceWorker = new Worker(
  MAINTENANCE_QUEUE,
  async (job) => {
    const startedAt = Date.now();
    logger.info({ platform: MAINTENANCE_PLATFORM, job: job.name }, 'job started');

    switch (job.name) {
      case MAINTENANCE_JOBS.DAILY_SUMMARY: {
        await writeDailySummary(new Date());
        logger.info({ platform: MAINTENANCE_PLATFORM, durationMs: Date.now() - startedAt }, 'daily summary written');
        break;
      }

      case MAINTENANCE_JOBS.CLEANUP_LOGS: {
        const deleted = await cleanupOldLogs();
        logger.info({ platform: MAINTENANCE_PLATFORM, deleted, durationMs: Date.now() - startedAt }, 'old sync_logs cleaned up');
        break;
      }

      default:
        throw new Error(`maintenanceWorker: unknown job name: ${job.name}`);
    }
  },
  { connection, concurrency: 1 },
);
```

> Maintenance jobs are fire-and-forget housekeeping — they do not call `logQueued` / `logRunning` / `logSuccess` / `logFailure`. They have their own queue and are observable via Bull Board.

### Step 8 — Delta Sync Verification

Verify the following pattern is consistent across **all 5 platform workers** before marking the phase done.

**Correct pattern (matches existing workers):**
```ts
// logQueued + logRunning BEFORE the try block
const queuedId = await logQueued(PLATFORM, job.name);
const syncLog  = await logRunning(queuedId);

try {
  const lastSyncedAt = await getLastSyncedAt(PLATFORM, job.name);
  const syncedAt = new Date();

  // ... fetch, transform, upsert ...

  await setLastSyncedAt(PLATFORM, job.name, syncedAt);  // ← BEFORE logSuccess
  await logSuccess(syncLog.id, { ... });
} catch (error) {
  // setLastSyncedAt NOT called — next run retries the same window
  await logFailure(syncLog.id, { ... });
  throw error;
}
```

**Full-refresh jobs** (Cin7 `INVENTORY` + `BRANCHES`, Shopify `INVENTORY`, GA4 `DAILY`, all Facebook insights jobs): `getLastSyncedAt` result is not passed to the fetch function but `setLastSyncedAt` is still called on success so the sync log shows it ran.

### Step 9 — Stuck Job Recovery

Run after any ungraceful shutdown to clear stuck `running` rows:

```sql
-- Find stuck jobs (running for > 1 hour)
SELECT id, platform, job_type, created_at
FROM sync_logs
WHERE status = 'running'
  AND created_at < NOW() - INTERVAL 1 HOUR;

-- Fix them
UPDATE sync_logs
SET   status        = 'failed',
      error_message = 'Process killed mid-job (ungraceful shutdown)',
      modified_at   = NOW()
WHERE status = 'running'
  AND created_at < NOW() - INTERVAL 1 HOUR;
```

> Uses `created_at` — the row insert timestamp. There is no `started_at` or `finished_at` column in `sync_logs`.
